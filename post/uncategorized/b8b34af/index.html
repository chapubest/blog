function debounce(fn, delay) {
  let timer;
  return function() {
    let that = this, args = arguments;
    if (timer) {
      clearTimeout(timer);
    }
    timer = setTimeout(function() {
      fn.apply(that, args);
    }, delay);
  }
}

function throttle(fn, delay) {
  let wait = false;
  return function() {
    let that = this, args = arguments;
    if(!wait) {
      wait = true;
      setTimeout(function() {
        fn.apply(that, args);
        wait = false;
      }, delay)
    }
  }
}
